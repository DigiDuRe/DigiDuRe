#Get author name / ID and birthyear

PREFIX schema: <http://schema.org/>

SELECT ?title_id ?author_id ?author (STR(?birthdate) AS ?birthdate_str) WHERE {
    ?title_id schema:author ?author_id .
    ?title_id schema:author/schema:name ?author .
    ?title_id schema:author/schema:birthDate ?birthdate .
    ?title_id a schema:Book .
    ?title_id schema:name ?title .
    ?title_id schema:mainEntityOfPage/schema:isPartOf <http://data.bibliotheken.nl/id/dataset/nbt> .
}
LIMIT 100
